```cpp
#include <iostream>
const int N = 550;

int main() {
    int T;
    std::cin >> T;
    while (T --) {
        int n;
        int a[N][N] = {0};
        int dp[N][N][2] = {0};  // dp[i][j][0] 表示小钟到达 a[i][j] 位置且不带走当前位置宝藏时的最大价值，dp[i][j][1] 表示小钟到达 a[i][j] 位置且带走当前位置宝藏的最大价值
        int res = 0;
        std::cin >> n;
        for (int i = 1; i <= n; ++ i) {
            for (int j = 1; j <= n; ++ j) {
                std::cin >> a[i][j];
            }
        }
        
        dp[1][1][0] = 0;
        dp[1][1][1] = a[1][1];
        for (int i = 1; i <= n + 1; ++ i) {
            for (int j = 1; j <= n + 1; ++ j) {
                dp[i][j][0] = std::max(dp[i][j - 1][1], dp[i + 1][j][0]);
                dp[i][j][1] = std::max(dp[i][j - 1][1], dp[i + 1][j][0]) + a[i][j];
            }
        }
        
        // 获得答案
        for (int i = 1; i <= n; ++ i) res = std::max(res, std::max(dp[n + 1][i][0], dp[n + 1][i][1]));
        for (int i = 1; i <= n; ++ i) res = std::max(res, std::max(dp[i][n + 1][0], dp[i][n + 1][1]));
        std::cout << res << std::endl;
    }
    return 0;
}
```
